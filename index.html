import { serve } from "https://deno.land/std@0.168.0/http/server.ts"; const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", }; serve(async (req) => { if (req.method === "OPTIONS") { return new Response("ok", { headers: corsHeaders }); } try { const { image, isCoppia } = await req.json(); const GEMINI_API_KEY = Deno.env.get("GEMINI_API_KEY"); if (!GEMINI_API_KEY) throw new Error("GEMINI_API_KEY non configurata"); const prompt = isCoppia ? `Analizza questa classifica di Nordic Walking. È una classifica A COPPIE. Estrai tutti i dati e rispondi SOLO con un array JSON (nessun markdown, nessuna spiegazione, solo il JSON puro): [{"pos":1,"p1":"Cognome Nome","p2":"Cognome Nome","assoc":"Nome ASD","tot":195,"pts":"195,3340876"}] IMPORTANTE: mantieni la virgola come separatore decimale nei punti. Includi TUTTE le righe visibili.` : `Analizza questa classifica di Nordic Walking. È una classifica INDIVIDUALE. Estrai tutti i dati e rispondi SOLO con un array JSON (nessun markdown, nessuna spiegazione, solo il JSON puro): [{"pos":1,"cognome":"Cognome","nome":"Nome","assoc":"Nome ASD","tot":195,"pts":"195,3340876"}] IMPORTANTE: mantieni la virgola come separatore decimale nei punti. Includi TUTTE le righe visibili.`; const resp = await fetch( `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${GEMINI_API_KEY}`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ contents: [{ parts: [ { inline_data: { mime_type: "image/jpeg", data: image } }, { text: prompt } ] }], generationConfig: { temperature: 0.1, maxOutputTokens: 4096 } }) } ); if (!resp.ok) { const err = await resp.json(); throw new Error(err.error?.message || `Gemini HTTP ${resp.status}`); } const data = await resp.json(); const text = data.candidates?.[0]?.content?.parts?.[0]?.text || ""; const cleaned = text.replace(/```json\n?/g, "").replace(/```\n?/g, "").trim(); const start = cleaned.indexOf("["); const end = cleaned.lastIndexOf("]"); if (start < 0 || end < 0) throw new Error("JSON non trovato nella risposta"); const rows = JSON.parse(cleaned.substring(start, end + 1)); return new Response(JSON.stringify({ rows }), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } catch (e) { return new Response(JSON.stringify({ error: e.message }), { status: 400, headers: { ...corsHeaders, "Content-Type": "application/json" } }); } });
